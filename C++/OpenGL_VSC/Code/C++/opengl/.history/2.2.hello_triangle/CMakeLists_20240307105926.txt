cmake_minimum_required(VERSION 3.27)

project(OPENGL_2.2.hello_triangle VERSION 1.0)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED TRUE)
set(INCLUDE_PATH "../Include")
set(LIB_PATH "../Libs")

include_directories(${INCLUDE_PATH})

link_libraries(${LIB_PATH})

add_executable(HelloTriangle hello_triangle.cpp)

target_link_libraries(HelloTriangle libglad.a libglfw3.a libglfw3dll.a)

if(WIN32)
    target_link_libraries(HelloTriangle opengl32)
endif()

add_custom_command(TARGET HelloTriangle POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    "${PROJECT_SOURCE_DIR}../output/glfw.dll"
    $<TARGET_FILE_DIR:HelloTriangle>)
