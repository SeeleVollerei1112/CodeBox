cmake_minimum_required(VERSION 3.27)

# 设置项目名称和版本
project(OpenGL_Project VERSION 1.0)

# 指定C++标准
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# 添加头文件目录
include_directories(../Include)

# 添加库文件目录
link_directories(../Libs)

# 添加可执行文件
add_executable(HelloWindow hello_window.cpp)

# 链接静态库文件
target_link_libraries(HelloWindow libglad.a libglfw3.a)

# 如果是在Windows上，可能还需要链接OpenGL库
if(WIN32)
  target_link_libraries(HelloWindow opengl32)
endif()

# 复制DLL到输出目录，如果是动态链接GLFW的话
add_custom_command(TARGET HelloWindow POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E copy_if_different
  "${PROJECT_SOURCE_DIR}/../output/glfw3.dll"
  $<TARGET_FILE_DIR:HelloWindow>)
