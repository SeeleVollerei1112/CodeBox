cmake_minimum_required(VERSION 3.27)

project(OPENGL_2.2.hello_triangle VERSION 1.0)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED TRUE)

# include_directories(../Include)
target_include_directories(HelloTriangle PUBLIC ${CMAKE_SOURCE_DIR}/../../Include)

# link_directories(../Libs)
target_include_directories(libglfw3dll.a PUBLIC ${CMAKE_SOURCE_DIR}/../../Libs)

add_executable(HelloTriangle hello_triangle.cpp)

target_link_libraries(HelloTriangle libglad.a libglfw3dll.a)

if(WIN32)
    target_link_libraries(HelloTriangle opengl32)
endif()

add_custom_command(TARGET HelloTriangle POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    "${PROJECT_SOURCE_DIR}/../output/glfw3.dll"
    $<TARGET_FILE_DIR:HelloTriangle>)
